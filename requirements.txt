# Global Collaborator - System Requirements
# ==========================================
#
# This is a Node.js project. Dependencies are managed via npm (package.json).
#
# SYSTEM REQUIREMENTS:
# - Node.js >= 18.0.0
# - npm >= 9.0.0
#
# INSTALLATION:
# Run: npm install
#
# This will install all dependencies listed in package.json:
# - react ^19.2.0
# - react-dom ^19.2.0
# - recharts ^3.6.0
# - tailwindcss ^4.1.18
# - vite ^7.2.4
# - And other dev dependencies...
#
# To start the development server:
# Run: npm run dev
#
# To build for production:
# Run: npm run build
